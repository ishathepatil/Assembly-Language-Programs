MESS MACRO MSG
MOV AH,09h
LEA DX,MSG
INT 21h
ENDM

.MODEL SMALL

.DATA
N1 DB 05h
MSG1 DB 0Ah,0dh, "ADDITION:$"
MSG2 DB 0Ah,0dh, "ENTER A NUMBER:$"
MSG3 DB 0Ah,0dh, "PRE-DEFINED NUMBER 05h:$"
MSG4 DB 0Ah,0dh, "ANSWER= :$"

.CODE
MOV AX,@DATA
MOV DS,AX

MESS MSG1
MESS MSG3
MESS MSG2
CALL ACCEPT
MOV AL,CL

ADD AL,N1

MESS MSG4
MOV CL,AL
CALL DISP

MOV AH,4CH
INT 21h

ACCEPT PROC NEAR
MOV AH,01h
INT 21h
CMP AL,3Ah
JC DOWN1
SUB AL,07h
DOWN1:AND AL,0Fh
ROR AL,04h
MOV BL,AL

MOV AH,01h
INT 21h
CMP AL,3Ah
JC DOWN2
SUB AL,07h
DOWN2:AND AL,0Fh
ADD BL,AL

MOV CL,BL

RET
ENDP

DISP PROC NEAR
MOV BL,CL
AND BL,0F0h
ROR BL,4h
CMP BL,0Ah
JC DOWN3
ADD BL,07h
DOWN3:ADD BL,30h
MOV DL,BL
MOV AH,02h
INT 21h

MOV BL,CL
AND BL,0Fh
CMP BL,0Ah
JC DOWN4
ADD BL,07h
DOWN4:ADD BL,30h
MOV DL,BL
MOV AH,02h
INT 21h
RET
ENDP

END
